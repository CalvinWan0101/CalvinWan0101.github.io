数独C++想法

有一天我去看了上清大资工同学的备审

奖项跟参赛记录多到哭出来啊，是完全无法触及的级别

但是他有提到他有用Excel写一个2*2的数独

刚好我最近在读关于dfs的东西，想说不然来玩玩看

逻辑应该差不多，就是比较不那麽暴力的穷举

正式开始

先建立一个全域的int阵列当做map

int map[9][9]

然后建立一个用来检测输入的数字是否符合条件的函数

条件如下:

范围 条件A 条件B
x相同的9个格子 格子内的数字互不相等 数字为1~9
y相同的9个格子 格子内的数字互不相等 数字为1~9
划分在同一个小格子的9个格子 格子内的数字互不相等 数字为1~9

这边的count是跑到第count格的意思

从第1格开始跑，到第(9*9)格结束

先计算当前count值下的坐标

int y = count / 9;
int x = count % 9;

检查y相同的状况，

for (int i = 0; i < 9; ++i)
	//当發现y相同时有点的值和代入的值相同时就return false
    if (map[y][i] == map[y][x] && i != x)
		return false;

检查x相同的状况

for (int i = 0; i < 9; ++i)
//当發现x相同时有点的值和代入的值相同时就return false
	if (map[i][x] == map[y][x] && i != y)
		return false;

检查同一小格的状况

//sx,sy指小格子的坐标
//x,y的范围是0~8,至于公式可以自己拿纸笔推推看
int sy = y / 3 * 3;
int sx = x / 3 * 3;
for (int i = sy; i < sy + 3; ++i)
	for (int k = sx; k < sx + 3; ++k)
		if (map[i][k] == map[y][x] && i != y && k != x)
			return false;

当到了这边就代表前面都没有return也就是这个值是没问题的

这时候就return true;

完整的Test函数长这样

bool Test(int count)
{
	int y = count / 9;
	int x = count % 9;
	//y相同
	for (int i = 0; i < 9; ++i)
		if (map[y][i] == map[y][x] && i != x)
			return false;
	//x相同
	for (int i = 0; i < 9; ++i)
		if (map[i][x] == map[y][x] && i != y)
			return false;
	//同一小格
	int sy = y / 3 * 3;
	int sx = x / 3 * 3;
	for (int i = sy; i < sy + 3; ++i)
		for (int k = sx; k < sx + 3; ++k)
			if (map[i][k] == map[y][x] && i != y && k != x)
				return false;
	return true;
}

然后是BackTracking的部分了(我这样玩好像有dfs的味道，但我不确定是不是dfs)

我把这个函数取名叫guess

第一步先判断是不是走到终点了

也就是count的值是否为81

if (count == 81)
	//如果是81就把map整个印出来再return
    {
		for (int i = 0; i < 9; ++i)
		{
			for (int j = 0; j < 9; ++j)
				cout << map[i][j] << " ";
			cout << endl;
		}
		return;
	}

确定了现在走的点不是最后一个点之后

这裡一样要计算count当下x,y的值

int y = count / 9;
int x = count % 9;

这时候分成两种情况

情况 动作
当前位置没有数字(map[y][x]==0) 开始在这一格做尝试
当前位置已有数字(map[y][x]!=0) 直接呼叫下一层

这边好像就比较有dfs的味道了

一个值试过之后当前格再重置为0继续试下一个值

当前位置没有数字时:

if (map[y][x] == 0)
	{
		for (int i = 1; i <= 9; ++i)
		{
			//给值标记已经使用过
			map[y][x] = i;
			//当检测合格的时候就再进到下一层
            if (Test(count))
				guess(count + 1);
		}
		//重置该点数值
		map[y][x] = 0;
	}

当前位置有数字时：

else
  //直接进入下一层
  guess(count + 1);

整个guess函数长这样：

void guess(int count)
{
	if (count == 81)
	{
		for (int i = 0; i < 9; ++i)
		{
			for (int j = 0; j < 9; ++j)
				cout << map[i][j] << " ";
			cout << endl;
		}
		return;
	}
	int y = count / 9;
	int x = count % 9;
	if (map[y][x] == 0)
	{
		for (int i = 1; i <= 9; ++i)
		{
			//给值标记已经使用过
			map[y][x] = i;
			if (Test(count))
				guess(count + 1);
		}
		//重置该点数值
		map[y][x] = 0;
	}
	else
		guess(count + 1);
}

最后一步就是main函数啦~

就只有输入map的值然后guess(0)

注意这边输入格式空格以0代替

int main()
{
	for (int i = 0; i < 9; ++i)
		for (int j = 0; j < 9; ++j)
			cin >> map[i][j];
	cout << endl;
	guess(0);
	return 0;
}

我是在图书馆把这一串给写完的

写完之后我就坐在那边开外挂完了半小时数独(刚好IG广告就是数独我吓到)

完整的程式码：

#include <iostream>
#include <algorithm>
using namespace std;
int map[9][9];
bool Test(int count)
{
	int y = count / 9;
	int x = count % 9;
	//同一行
	for (int i = 0; i < 9; ++i)
		if (map[y][i] == map[y][x] && i != x)
			return false;
	//同一列
	for (int i = 0; i < 9; ++i)
		if (map[i][x] == map[y][x] && i != y)
			return false;
	//同一小格
	int sy = y / 3 * 3;
	int sx = x / 3 * 3;
	for (int i = sy; i < sy + 3; ++i)
		for (int k = sx; k < sx + 3; ++k)
			if (map[i][k] == map[y][x] && i != y && k != x)
				return false;
	return true;
}
void guess(int count)
{
	if (count == 81)
	{
		for (int i = 0; i < 9; ++i)
		{
			for (int j = 0; j < 9; ++j)
				cout << map[i][j] << " ";
			cout << endl;
		}
		return;
	}
	int y = count / 9;
	int x = count % 9;
	if (map[y][x] == 0)
	{
		for (int i = 1; i <= 9; ++i)
		{
			//给值标记已经使用过
			map[y][x] = i;
			if (Test(count))
				guess(count + 1);
		}
		//重置该点数值
		map[y][x] = 0;
	}
	else
		guess(count + 1);
}
int main()
{
	for (int i = 0; i < 9; ++i)
		for (int j = 0; j < 9; ++j)
			cin >> map[i][j];
	cout << endl;
	guess(0);
	return 0;
}
